## Step 1: Create Your First Behavior (1 minute)

Create a file called `counter.ts`:

```typescript
// Define your state shape
interface CounterState {
  count: number;
}

// Create a component that takes a slice factory
export const createCounter = (createSlice) => {
  const counter = createSlice(({ get, set }) => ({
    // Getters return current values
    count: () => get().count,
    
    // Actions update state
    increment: () => set({ count: get().count + 1 }),
    decrement: () => set({ count: get().count - 1 }),
    reset: () => set({ count: 0 })
  }));
  
  return { counter };
};
```

<Aside type="tip">
  That's it! You've just created a portable counter behavior. This same code will work in React, Vue, Svelte, and vanilla JavaScript.
</Aside>

## Step 2: Choose Your Adapter (1 minute)

Lattice works with your existing state management library. Pick the one you're already using:

<Tabs>
  <TabItem label="Zustand">
    ```bash
    npm install @lattice/adapter-zustand zustand
    ```
    ```typescript
    import { createStore } from '@lattice/adapter-zustand';
    import { createCounter } from './counter';

    // Create a store and pass it to your component
    const createSlice = createStore({ count: 0 });
    export const { counter } = createCounter(createSlice);
    ```

## Step 3: Use in Your Framework (3 minutes)

Now use your counter in any framework:

<Tabs>
  <TabItem label="React">
    ```tsx
    import { useSliceSelector } from '@lattice/runtime/react';
    import { counter } from './store';

    function Counter() {
      const count = useSliceSelector(counter, s => s.count());
      const { increment, decrement, reset } = counter.selector;
      
      return (
        <div>
          <h1>Count: {count}</h1>
          <button onClick={increment}>+</button>
          <button onClick={decrement}>-</button>
          <button onClick={reset}>Reset</button>
        </div>
      );
    }
    ```
  </TabItem>
  <TabItem label="Vue">
    ```vue
    <script setup>
    import { useSlice } from '@lattice/runtime/vue';
    import { counter } from './store';

    const count = useSlice(counter, s => s.count());
    const { increment, decrement, reset } = counter.selector;
    </script>

    <template>
      <div>
        <h1>Count: {{ count }}</h1>
        <button @click="increment">+</button>
        <button @click="decrement">-</button>
        <button @click="reset">Reset</button>
      </div>
    </template>
    ```
  </TabItem>
  <TabItem label="Svelte">
    ```svelte
    <script>
    import { createStore } from '@lattice/adapter-svelte';
    import { createCounter } from './counter';
    
    // Svelte can use reactive state directly
    const state = $state({ count: 0 });
    const createSlice = createStore(state);
    const { counter } = createCounter(createSlice);
    
    const { increment, decrement, reset } = counter.selector;
    </script>

    <div>
      <h1>Count: {state.count}</h1>
      <button on:click={increment}>+</button>
      <button on:click={decrement}>-</button>
      <button on:click={reset}>Reset</button>
    </div>
    ```
  </TabItem>
  <TabItem label="Vanilla JS">
    ```javascript
    import { counter } from './store';

    const { count, increment, decrement, reset } = counter.selector;

    // Get current value
    console.log(count()); // 0

    // Update state
    increment();
    console.log(count()); // 1

    // Subscribe to changes
    const unsubscribe = counter.subscribe(() => {
      console.log('Count changed:', count());
    });
    ```
  </TabItem>
</Tabs>
